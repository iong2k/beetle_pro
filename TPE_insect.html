#"""為overpass request的格式，內容為overpass語法:"""[out:資料格式]；[空間向量資料=標籤]；(空間範圍)；out；/變數

data = response.json() #以JSON檔讀取的回應

m = folium.Map(location=[25.0462394, 121.5309870], zoom_start=12) #m為地圖，.Map用來生成地圖(定義中心坐標，再設定尺度zoom_start，數字和比例成正比)

for element in data['elements']: #for...in為迴圈、element為個別參數
    lat = element['lat']
    lon = element['lon']
    name = element.get('tags', {}).get('name')
    #上述為資料定義
    folium.Marker(
        location=[lat, lon],
        popup=name,
        icon=folium.Icon(icon='info-sign')
    ).add_to(m)
